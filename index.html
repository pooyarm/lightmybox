LightMyBox

Default Style

$(".lightmybox").lightmybox();

Solid Background Color (Auto Mode)

$(".lightmybox").lightmybox({
	bg: 'color',
	color: 'auto'
});

Solid Background Color (Defined Color)

$(".lightmybox").lightmybox({
	bg: 'color',
	color: '#e53737'
});

Pixelate Background Mode

$(".lightmybox").lightmybox({
	bg: 'pixelate'
});

Single Mode

$(".lightmybox").lightmybox({
	mode: 'single'
});

Light Theme

$(".lightmybox").lightmybox({
	theme: 'light'
});

Without Frame Without Opacity

$(".lightmybox").lightmybox({
	bgOpacity: 1,
	frame: false
});

Fullscreen Mode

$(".lightmybox").lightmybox({
	height: 'fullscreen'
});
